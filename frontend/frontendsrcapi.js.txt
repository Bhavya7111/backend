import React, { useEffect, useState } from "react";
import { getBackendMessage } from "./api";

function App() {
  const [message, setMessage] = useState("");

  useEffect(() => {
    getBackendMessage().then(setMessage);
  }, []);

  return (
    <div>
      <h1>DevOps Practice Project</h1>
      <p>{message}</p>
    </div>
  );
}

export default App;
